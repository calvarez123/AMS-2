Para que puedas leer y escribir en el mismo archivo, necesitas realizar el proceso de lectura y escritura en un archivo temporal y luego reemplazar el archivo original con el archivo temporal. Aquí te muestro cómo hacerlo:

java
import java.io.*;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        String filePathIn = "Excercici1.java"; // Ruta del archivo de entrada
        String filePathOut = "temp.txt"; // Ruta de un archivo temporal

        try {
            FileWriter fOut = new FileWriter(filePathOut);
            File file = new File(filePathIn);
            Scanner scnr = new Scanner(file);

            while (scnr.hasNextLine()) {
                String line = scnr.nextLine();
                // Realiza alguna manipulación en 'line' si es necesario
                // Por ejemplo, puedes invertir el contenido de la línea
                // Aquí, simplemente copiaremos la línea sin cambios.
                fOut.write(line + "\n");
            }

            scnr.close();
            fOut.close();

            // Reemplaza el archivo original con el archivo temporal
            File originalFile = new File(filePathIn);
            File tempFile = new File(filePathOut);

            if (originalFile.delete()) {
                if (tempFile.renameTo(originalFile)) {
                    System.out.println("Archivo actualizado con éxito.");
                } else {
                    System.out.println("No se pudo reemplazar el archivo original.");
                }
            } else {
                System.out.println("No se pudo eliminar el archivo original.");
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}

Para leer y procesar el archivo Exercici1.java y escribirlo en el mismo archivo, puedes hacerlo de la siguiente manera. Ten en cuenta que debes tomar precauciones al hacer esto, ya que sobrescribirás el archivo fuente. java
import java.io.*;

public class Exercici1 {
    public static void main(String args[]) {
        String basePathIn = System.getProperty("user.dir") + "/src/main/java/cat/iesesteveterradas/fites/";
        String filePathIn = basePathIn + "Exercici1.java";

        try {
            BufferedReader reader = new BufferedReader(new FileReader(filePathIn));
            StringBuilder fileContent = new StringBuilder();
            String line;

            while ((line = reader.readLine()) != null) {
                fileContent.append(line).append("\n");
            }
            reader.close();

            String code = fileContent.toString();
            String invertedCode = giraText(code);

            FileWriter writer = new FileWriter(filePathIn);
            writer.write(invertedCode);
            writer.close();
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    public static String giraText(String text) {
        String resultat = "";
        for (int cnt = text.length() - 1; cnt >= 0; cnt--) {
            resultat = resultat + text.charAt(cnt);
        }
        return resultat;
    }
}
